    "class SyntheticCreditCardDataGenerator:\n",
    "    \"\"\"\n",
    "    Generates synthetic credit card application data with configurable parameters.\n",
    "    \n",
    "    This class creates realistic credit card application datasets that can be used\n",
    "    for machine learning demonstrations, particularly for credit underwriting models.\n",
    "    \"\"\"\n",
    "    \n",
    "    def __init__(self, \n",
    "                 size=10000,\n",
    "                 locale='en_IN',\n",
    "                 age_range=(21, 65),\n",
    "                 income_range=(200000, 5000000),\n",
    "                 approval_rate=0.7,\n",
    "                 seed=42):\n",
    "        \"\"\"\n",
    "        Initialize the data generator with configurable parameters.\n",
    "        \n",
    "        Parameters:\n",
    "        -----------\n",
    "        size : int\n",
    "            Number of records to generate (default: 10000)\n",
    "        locale : str\n",
    "            Faker locale for generating names and addresses (default: 'en_IN')\n",
    "        age_range : tuple\n",
    "            Min and max age range (default: (21, 65))\n",
    "        income_range : tuple\n",
    "            Min and max annual income in INR (default: (200000, 5000000))\n",
    "        approval_rate : float\n",
    "            Target approval rate (0.0 to 1.0, default: 0.7)\n",
    "        seed : int\n",
    "            Random seed for reproducibility (default: 42)\n",
    "        \"\"\"\n",
    "        self.size = size\n",
    "        self.locale = locale\n",
    "        self.age_range = age_range\n",
    "        self.income_range = income_range\n",
    "        self.approval_rate = approval_rate\n",
    "        self.seed = seed\n",
    "        \n",
    "        # Initialize Faker and set seeds for reproducibility\n",
    "        self.fake = Faker(locale)\n",
    "        Faker.seed(seed)\n",
    "        np.random.seed(seed)\n",
    "        random.seed(seed)\n",
    "        \n",
    "        print(f\"Initialized generator for {size} records with locale {locale}\")\n",
    "    \n",
    "    def _generate_customer_demographics(self):\n",
    "        \"\"\"Generate basic customer demographic information.\"\"\"\n",
    "        customer_ids = [f\"CUST{i:08d}\" for i in range(1, self.size + 1)]\n",
    "        names = [self.fake.name() for _ in range(self.size)]\n",
    "        ages = np.random.randint(self.age_range[0], self.age_range[1] + 1, self.size)\n",
    "        genders = np.random.choice(['M', 'F'], self.size, p=[0.52, 0.48])\n",
    "        \n",
    "        return customer_ids, names, ages, genders\n",
    "    \n",
    "    def _generate_financial_data(self):\n",
    "        \"\"\"Generate financial information for customers.\"\"\"\n",
    "        # Annual income with log-normal distribution (more realistic)\n",
    "        mu = np.log(800000)  # Median income around 8 lakhs\n",
    "        sigma = 0.8\n",
    "        incomes = np.random.lognormal(mu, sigma, self.size)\n",
    "        incomes = np.clip(incomes, self.income_range[0], self.income_range[1])\n",
    "        incomes = incomes.astype(int)\n",
    "        \n",
    "        # Employment status\n",
    "        employment_statuses = np.random.choice(\n",
    "            ['Salaried', 'Self-Employed', 'Business', 'Professional', 'Retired'],\n",
    "            self.size,\n",
    "            p=[0.6, 0.2, 0.1, 0.08, 0.02]\n",
    "        )\n",
    "        \n",
    "        # Credit score (300-850 range, normal distribution)\n",
    "        credit_scores = np.random.normal(680, 80, self.size)\n",
    "        credit_scores = np.clip(credit_scores, 300, 850).astype(int)\n",
    "        \n",
    "        # Existing debt (correlated with income)\n",
    "        debt_ratio = np.random.beta(2, 5, self.size)  # Most people have lower debt ratios\n",
    "        existing_debts = (incomes * debt_ratio * np.random.uniform(0.1, 0.8, self.size)).astype(int)\n",
    "        \n",
    "        # Number of existing credit cards\n",
    "        num_cards = np.random.poisson(1.5, self.size)\n",
    "        num_cards = np.clip(num_cards, 0, 8)\n",
    "        \n",
    "        # Monthly salary (for salaried employees)\n",
    "        monthly_salaries = (incomes / 12).astype(int)\n",
    "        \n",
    "        return incomes, employment_statuses, credit_scores, existing_debts, num_cards, monthly_salaries\n",
    "    \n",
    "    def _generate_application_data(self):\n",
    "        \"\"\"Generate credit card application specific data.\"\"\"\n",
    "        # Requested credit limit (typically 2-5x monthly income)\n",
    "        monthly_incomes = np.array([inc/12 for inc in self._temp_incomes])\n",
    "        multipliers = np.random.uniform(2, 8, self.size)\n",
    "        requested_limits = (monthly_incomes * multipliers).astype(int)\n",
    "        requested_limits = np.clip(requested_limits, 15000, 500000)\n",
    "        \n",
    "        # Application dates (last 2 years)\n",
    "        start_date = datetime.now() - timedelta(days=730)\n",
    "        end_date = datetime.now()\n",
    "        application_dates = [\n",
    "            self.fake.date_between(start_date=start_date, end_date=end_date)\n",
    "            for _ in range(self.size)\n",
    "        ]\n",
    "        \n",
    "        # Card type preference\n",
    "        card_types = np.random.choice(\n",
    "            ['Standard', 'Gold', 'Platinum', 'Premium'],\n",
    "            self.size,\n",
    "            p=[0.4, 0.3, 0.2, 0.1]\n",
    "        )\n",
    "        \n",
    "        return requested_limits, application_dates, card_types\n",
    "    \n",
    "    def _determine_approval_and_limit(self, credit_scores, incomes, existing_debts, requested_limits, ages, employment_statuses):\n",
    "        \"\"\"\n",
    "        Determine approval status and credit limit based on underwriting rules.\n",
    "        \n",
    "        This method implements realistic credit underwriting logic:\n",
    "        - Credit score is the primary factor\n",
    "        - Income-to-debt ratio is considered\n",
    "        - Age and employment status influence decisions\n",
    "        - Requested limit vs. income ratio matters\n",
    "        \"\"\"\n",
    "        approved_status = []\n",
    "        approved_limits = []\n",
    "        \n",
    "        for i in range(self.size):\n",
    "            score = credit_scores[i]\n",
    "            income = incomes[i]\n",
    "            debt = existing_debts[i]\n",
    "            requested = requested_limits[i]\n",
    "            age = ages[i]\n",
    "            employment = employment_statuses[i]\n",
    "            \n",
    "            # Calculate debt-to-income ratio\n",
    "            dti_ratio = debt / income if income > 0 else 1\n",
    "            \n",
    "            # Calculate approval probability based on multiple factors\n",
    "            prob_approve = 0.0\n",
    "            \n",
    "            # Credit score factor (40% weight)\n",
    "            if score >= 750:\n",
    "                prob_approve += 0.4\n",
    "            elif score >= 700:\n",
    "                prob_approve += 0.32\n",
    "            elif score >= 650:\n",
    "                prob_approve += 0.24\n",
    "            elif score >= 600:\n",
    "                prob_approve += 0.16\n",
    "            elif score >= 550:\n",
    "                prob_approve += 0.08\n",
    "            \n",
    "            # Income factor (25% weight)\n",
    "            if income >= 1000000:\n",
    "                prob_approve += 0.25\n",
    "            elif income >= 600000:\n",
    "                prob_approve += 0.20\n",
    "            elif income >= 400000:\n",
    "                prob_approve += 0.15\n",
    "            elif income >= 250000:\n",
    "                prob_approve += 0.10\n",
    "            \n",
    "            # Debt-to-income factor (20% weight)\n",
    "            if dti_ratio <= 0.3:\n",
    "                prob_approve += 0.20\n",
    "            elif dti_ratio <= 0.5:\n",
    "                prob_approve += 0.15\n",
    "            elif dti_ratio <= 0.7:\n",
    "                prob_approve += 0.10\n",
    "            \n",
    "            # Employment factor (10% weight)\n",
    "            if employment in ['Salaried', 'Professional']:\n",
    "                prob_approve += 0.10\n",
    "            elif employment == 'Self-Employed':\n",
    "                prob_approve += 0.08\n",
    "            elif employment == 'Business':\n",
    "                prob_approve += 0.06\n",
    "            \n",
    "            # Age factor (5% weight)\n",
    "            if 25 <= age <= 55:\n",
    "                prob_approve += 0.05\n",
    "            elif age > 55:\n",
    "                prob_approve += 0.03\n",
    "            \n",
    "            # Make approval decision\n",
    "            if random.random() < prob_approve:\n",
    "                approved_status.append('Yes')\n",
    "                \n",
    "                # Determine approved limit (usually 70-100% of requested)\n",
    "                if score >= 750:\n",
    "                    limit_ratio = random.uniform(0.85, 1.0)\n",
    "                elif score >= 700:\n",
    "                    limit_ratio = random.uniform(0.75, 0.95)\n",
    "                elif score >= 650:\n",
    "                    limit_ratio = random.uniform(0.65, 0.85)\n",
    "                else:\n",
    "                    limit_ratio = random.uniform(0.50, 0.75)\n",
    "                \n",
    "                approved_limit = int(requested * limit_ratio)\n",
    "                \n",
    "                # Cap the limit based on income (max 3x monthly income)\n",
    "                max_limit = int(income / 12 * 3)\n",
    "                approved_limit = min(approved_limit, max_limit)\n",
    "                \n",
    "                approved_limits.append(approved_limit)\n",
    "            else:\n",
    "                approved_status.append('No')\n",
    "                approved_limits.append(0)\n",
    "        \n",
    "        return approved_status, approved_limits\n",
    "    \n",
    "    def generate_dataset(self, output_file='syntheticdata.csv'):\n",
    "        \"\"\"\n",
    "        Generate the complete synthetic dataset and save to CSV.\n",
    "        \n",
    "        Parameters:\n",
    "        -----------\n",
    "        output_file : str\n",
    "            Output CSV filename (default: 'syntheticdata.csv')\n",
    "            \n",
    "        Returns:\n",
    "        --------\n",
    "        pandas.DataFrame\n",
    "            Generated dataset\n",
    "        \"\"\"\n",
    "        print(\"Starting dataset generation...\")\n",
    "        \n",
    "        # Generate all data components\n",
    "        customer_ids, names, ages, genders = self._generate_customer_demographics()\n",
    "        incomes, employment_statuses, credit_scores, existing_debts, num_cards, monthly_salaries = self._generate_financial_data()\n",
    "        \n",
    "        # Store incomes temporarily for application data generation\n",
    "        self._temp_incomes = incomes\n",
    "        requested_limits, application_dates, card_types = self._generate_application_data()\n",
    "        \n",
    "        # Determine approval and limits\n",
    "        approved_status, approved_limits = self._determine_approval_and_limit(\n",
    "            credit_scores, incomes, existing_debts, requested_limits, ages, employment_statuses\n",
    "        )\n",
    "        \n",
    "        # Create DataFrame\n",
    "        data = {\n",
    "            'CustomerID': customer_ids,\n",
    "            'Name': names,\n",
    "            'Age': ages,\n",
    "            'Gender': genders,\n",
    "            'AnnualIncome': incomes,\n",
    "            'MonthlySalary': monthly_salaries,\n",
    "            'EmploymentStatus': employment_statuses,\n",
    "            'CreditScore': credit_scores,\n",
    "            'ExistingDebt': existing_debts,\n",
    "            'NumCreditCards': num_cards,\n",
    "            'RequestedCreditLimit': requested_limits,\n",
    "            'CardType': card_types,\n",
    "            'ApplicationDate': application_dates,\n",
    "            'Approved': approved_status,\n",
    "            'ApprovedCreditLimit': approved_limits\n",
    "        }\n",
    "        \n",
    "        df = pd.DataFrame(data)\n",
    "        \n",
    "        # Save to CSV\n",
    "        df.to_csv(output_file, index=False)\n",
    "        \n",
    "        # Print summary statistics\n",
    "        approval_rate = (df['Approved'] == 'Yes').mean()\n",
    "        avg_approved_limit = df[df['Approved'] == 'Yes']['ApprovedCreditLimit'].mean()\n",
    "        \n",
    "        print(f\"Dataset generated successfully!\")\n",
    "        print(f\"Total records: {len(df)}\")\n",
    "        print(f\"Approval rate: {approval_rate:.2%}\")\n",
    "        print(f\"Average approved limit: ₹{avg_approved_limit:,.0f}\")\n",
    "        print(f\"Data saved to: {output_file}\")\n",
    "        \n",
    "        return df\n",
    "\n",
    "print(\"Data generator class defined successfully!\")"
